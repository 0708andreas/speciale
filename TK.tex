  \settowidth{\@rbll}{\hbox{#2}}%
  #2\kern-#1\@rbll#2}

    $\,\!$%
    \raisebox{-0.6pt}{M}%
    \rotatebox{8}{\rbl{.94}{M}}%
    ER\hspace{1pt}}}%

}}
